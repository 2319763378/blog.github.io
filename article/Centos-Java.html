安装Java环境

我使用的是Centos7.4

1. 下载

[root@server ~]# mkdir /usr/java/ && cd /usr/java/
[root@server java]# wget https://javadl.oracle.com/webapps/download/AutoDL?BundleId=236878_42970487e3af4f5aa5bca3f542482c60


[root@server java]# tar zxf jre-8u201-linux-x64.tar.gz

2. 安装

[root@server java]# vim /etc/profile

加入一下内容

 export JAVA_HOME=/usr/java/jre1.8.0_201
 export CLASSPATH=$CLASSPATH:$JAVA_HOME/lib:$JAVA_HOME/jre/lib
 export PATH=$JAVA_HOME/bin/:$JAVA_HOME/jre/bin/:$PATH:$HOME/bin
[root@server java]# source /etc/profile

3. 测试

[root@server java]# java -version
java version "1.8.0_201"
Java(TM) SE Runtime Environment (build 1.8.0_201-b09)
Java HotSpot(TM) 64-Bit Server VM (build 25.201-b09, mixed mode)
