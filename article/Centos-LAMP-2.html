我的环境是Centos 7.4，计上.

配置Discuz

1. 下载Discuz源码

[root@server ~]# cd /usr/src/
[root@server src]# wget http://download.comsenz.com/DiscuzX/3.1/Discuz_X3.1_SC_UTF8.zip
[root@server src]# unzip Discuz_X3.1_SC_UTF8.zip -d /usr/local/apache/htdocs/
[root@server src]# cd /usr/local/apache/htdocs/
[root@server htdocs]# mv upload/* .
[root@server htdocs]# chmod 757 -R data/ uc_server/ config/ uc_client/

2. 创建mysql及用户

[root@server htdocs]# mysql
 mysql> create database discuz charset=utf8;
 Query OK, 1 row affected (0.01 sec)
 mysql> grant all on discuz.* to 'root'@'localhost' identified by "123456";
 Query OK, 0 rows affected (0.00 sec)
 mysql> exit
 Bye
[root@server htdocs]#
